    $ cd myproject

# The .git Directory

    $ tree -a
    .
    ├── .git
    │   ├── config
    │   ├── HEAD
    │   ├── hooks
    │   │   ├── ctags
    │   │   ├── post-checkout -> post-commit
    │   │   ├── post-commit
    │   │   ├── post-merge -> post-commit
    │   │   └── post-rewrite
    │   ├── objects
    │   │   ├── info
    │   │   └── pack
    │   └── refs
    │       ├── heads
    │       └── tags
    └── README
